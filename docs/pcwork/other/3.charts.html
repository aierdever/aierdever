框架中集成了 echarts ,详细的echarts API及实例请 [参考这里](Apache EChartsopen in new window)

# echarts 使用示例代码

js:

define(['echarts','echarts.theme','pub'] ,function(echarts){
    var piebox = echarts.init('pieBox','v');
    piebox.setOption({
        title: {
            text: '各省区累积患者占比',
            // subtext: '纯属虚构',
            left: 'center',
            top:10
        },
        tooltip: {
            trigger: 'item',
            formatter: '{a} <br/>{b} : {c} ({d}%)'
        },
        series: [{
            name: '访问来源',
            type: 'pie',
            radius: '70%',
            center: ['50%', '56%'],
            label: {
                show: false
            },
            data: [{
                value: 335,
                name: '直接访问'
            },
            {
                 value: 310,
                 name: '邮件营销'
             },
             {
                 value: 234,
                 name: '联盟广告'
             },
             {
                  value: 135,
                  name: '视频广告'
              },
              {
                  value: 1548,
                  name: '搜索引擎'
              }],
        }]
    }, true);
});
